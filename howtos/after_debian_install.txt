AFTER DEBIAN INSTALL

# login as root and update debian
apt-get update -y && apt-get upgrade

# install some essential package: sudo, bc, ssh, git, mc, curl
apt-get install sudo bc ssh git mc curl -y

# create admin
adduser admin

# add admin to sudo
adduser admin sudo

# login as admin
su admin --login

#prepare for cloning
cd ~

# clone debian_scripts
sudo git clone https://github.com/zsbali/debian_scripts.git

# clone ec2-api-tools
sudo git clone https://github.com/traviscross/ec2-api-tools.git
